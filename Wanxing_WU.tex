% ── PUBLICATIONS ─────────────────────────────────────────────
\section{Publications}

\pubh{Towards Fair and Comprehensive Evaluation of Routers in Collaborative LLM Systems}{\troute}
    {\me{Wanxing Wu}, He Zhu, Yixia Li, \ldots, Guanhua Chen\corr}
    {Under Review, EMNLP 2026 \quad[\href{https://arxiv.org/abs/2602.11877}{arXiv:2602.11877}]}
    {Built \textbf{RouterXBench}, a 3-axis evaluator (AUROC / LPM-MPM-HCR / 6 ID-OOD benchmarks), and used it to show that \textbf{internal hidden states are a more discriminative routing signal} than verbose, logit, and external-embedding alternatives, with the finding holding across small models of different families and sizes.}

\pubh{Can Weak Models Explore? Scope Then Cover for LLM Agent Adaptation}{\trl}
    {\me{Wanxing Wu}\eqstar, He Zhu\eqstar, Guanhua Chen\corr \quad ($^*$Equal contribution)}
    {Under Review, 2026}
    {Adapting an LLM agent to a new environment hinges on exploration. Two controlled probes surfaced two recurring phenomena: scaling the explorer model brings no gain when the environment is uninformative about target location, and all scales collapse to random search once the candidate space outgrows the per-step inspection budget. Both reveal the same structure inside exploration: scoping a promising subspace is capability-bound, while verifying whether the target lies inside it is not. We propose \textbf{scope-then-cover}, where a capable actor nominates a small high-recall subspace and frozen cheap workers cover it in parallel, exposed as a single \texttt{explore(scope, query)} action. The same action drops into both a frozen test-time actor and an online experiential learning host, and lifts pass rate over no-explore baselines on ALFWorld, WebShop, and SWE-Bench-Verified.}

\pubh{AutoInterpBench: Level-1 Grounded Benchmarking for Autonomous Mechanistic Interpretability Discovery}{\tautoresearch}
    {Xuanzhe Xu, He Zhu, \me{Wanxing Wu}, Guanhua Chen\corr}
    {Under Review, EMNLP 2026}
    {Built \textbf{AutoInterpBench}, a three-level benchmark (264 tasks, 764 GPU runs, 14 models) for \emph{autoresearch} agents on mechanistic interpretability, paired with a \textbf{harness-driven discovery} loop that uses Level-1 scores as fitness signals to refine the agent's analysis primitives.}
